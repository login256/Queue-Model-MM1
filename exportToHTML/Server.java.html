1    import java.util.Random;

3    public class Server implements Comparable<Server> {
4        private static Random random = new Random();
5        public static double mu;
6        private static int cntId = 0;
7        public int id;
8        public boolean busy = false;
9        private Customer curCustomer;
10       public double freeTime = 0;
11   
12       private double generateTime() {
13           return -Math.log(random.nextDouble()) * mu; // - ln(rand()) / lambda
14       }
15   
16       public void serve(Customer customer, double curTime) {
17           customer.serve_time = curTime;
18           busy = true;
19           curCustomer = customer;
20           freeTime = curTime + generateTime();
21           System.out.println(curTime + ": 窗口 " + id + " 开始了 对顾客 " + curCustomer.id + " 的服务");
22       }
23   
24       public void update(double curTime) {
25           if (busy && freeTime < curTime) {
26               busy = false;
27               System.out.println(freeTime + ": 窗口 " + id + " 结束了 对顾客 " + curCustomer.id + " 的服务");
28               curCustomer = null;
29           }
30       }
31   
32       public Server() {
33           cntId++;
34           id = cntId;
35       }
36   
37       @Override
38       public int compareTo(Server o) {
39           if (!this.busy && !o.busy) {
40               return Integer.compare(this.id, o.id);
41           } else if (!this.busy && o.busy) {
42               return -1;
43           } else if (this.busy && !o.busy) {
44               return 1;
45           } else {
46               int k = Double.compare(this.freeTime, o.freeTime);
47               return k != 0 ? k : Integer.compare(this.id, o.id);
48           }
49       }
50   }
